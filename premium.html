async function initiatePayment() { const btn = document.querySelector('.btn-pay'); btn.innerText = "მიმდინარეობს..."; btn.disabled = true; try { const response = await fetch('/api/payment', { method: 'POST' }); const data = await response.json(); if (data.url) { window.location.href = data.url; // გადავიყვანთ ბანკის გვერდზე } else { alert("შეცდომა გადახდისას. სცადეთ მოგვიანებით."); btn.disabled = false; } } catch (e) { alert("კავშირის შეცდომა."); btn.disabled = false; } }
